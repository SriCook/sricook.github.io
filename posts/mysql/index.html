mysql命令 #

连接至数据库

mysql -h <ip> -u <username> -p

数据库操作 #

显示所有数据库

show databases;

切换数据库

use <dbname>;

创建数据库

create database <dbname>;

删除数据库

drop database <dbname>;

表操作 #

用一个表更新另一个表的若干列

update <table1> i, <table2> j
set i.<column1>=j.<column1>, i.<column2>=j.<column2>
where i.<id>=j.<id>;
